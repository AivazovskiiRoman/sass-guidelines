{% highlight scss %} /// Υπολόγισε το μέγιστο βάθος του map /// @param {Map} $map /// @return {Number} μέγιστο βάθος του `$map` @function map-depth($map) { $level: 1; @each $key, $value in $map { @if type-of($value) == 'map' { $level: max(map-depth($value) + 1, $level); } } @return $level; } {% endhighlight %}
